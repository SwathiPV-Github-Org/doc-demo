import { HttpClient, HttpParams } from '@angular/common/http';
import { ServerDataSource } from 'ng2-smart-table';
import cloneDeep from 'lodash/cloneDeep';
import isEqual from 'lodash/isEqual';
import isEmpty from 'lodash/isEmpty';
import { Observable, catchError, of } from 'rxjs';

export class CustomServerDataSource extends ServerDataSource {
  // Custom loader to be able to show the loader on pagination and filter actions
  tableLoader: boolean = true;
  prevConfig: {};
  constructor(
    protected http: HttpClient,
    conf: {} = {},
    private customOptions: any = {},
    private praseResponseData,
    private incrementPageNumber: boolean = false,
    private errorHandler?,
  ) {
    super(http, conf);
  }
  protected extractDataFromResponse(res: any): Array<any> {
    /**
     * By default we don't have control over setting some custom information before rendering into table
     * With this, we will be able to pass the callback from component and
     * in the callback we can all data manipulation needed before rendering into table.
     * For ex: in releases page, we need to 'compareUrl' field for some of the deployment row
     */
    const response = super.extractDataFromResponse(res);
    const parsedResponse = this.praseResponseData(
      cloneDeep(response),
      this.filterConf?.filters,
    );
    this.tableLoader = false;

    return parsedResponse;
  }
  protected addPagerRequestParams(httpParams: HttpParams): HttpParams {
    /* By default, ng2-smart-table sends the page being 1, but from BE to get first set of data we need to send -1
    Hence, reducing the page value by 1 here*/
    this.tableLoader = true;
    if (
      this.pagingConf &&
      this.pagingConf['page'] &&
      this.pagingConf['perPage']
    ) {
      httpParams = httpParams.set(
        this.conf.pagerPageKey,
        this.incrementPageNumber
          ? this.pagingConf['page']
          : this.pagingConf['page'] - 1,
      );
      httpParams = httpParams.set(
        this.conf.pagerLimitKey,
        this.pagingConf['perPage'],
      );
    }
    httpParams = this.addCustomQueryParams(httpParams);

    return httpParams;
  }
  /* Sometimes we need to send the queryparams from outside of the table interaction
  (For ex: Show No Change toggle in releases page)
  Using this function, you can set the custom query params to httpParams */
  addCustomQueryParams(httpParams: HttpParams): HttpParams {
    const { customQueryParams } = this.customOptions || {};
    if (customQueryParams) {
      let queryParamValue = '';
      Object.entries(customQueryParams || {})?.forEach(
        ([key, value]: [string, string | []]) => {
          queryParamValue = Array.isArray(value) ? value?.join(',') : value;
          if (queryParamValue) {
            httpParams = httpParams.set(key, queryParamValue);
          }
        },
      );
    }
    return httpParams;
  }
  async getElements(): Promise<any> {
    /** Overriding the default getElements method here to fix one issue: [https://github.com/akveo/ng2-smart-table/issues/237 - Still Open]
     * The issue is that, if user applies some filters on the table,
     * and if we need to reset the filters and reload the table without filters/pagination programatically without interacting with table
     * (Example: in releases page, if the latest release is in progress, then we reload the table)
     * then table automatically makes multiple api calls if any of the filters in table are interacted by user
     */

    this.createRequesParams(); // Need to calls this function to be able to get the new applied filters, sorting, pagination params info

    const currentConfig = {
      ...this.sortConf,
      ...this.pagingConf,
      ...this.filterConf,
    };

    /* When we reload the table with default pagination & filter, we re-instatiate the ServerDataSource,
      After re-instantiating, first call made will have empty filters array in this.filterConf,
      but if user had interacted with table filters before the table reload,
      then filters array will be present but the search value will be empty, due to which multiple API calls are made one for each filter.
      Custom logic has been written here to check if there are no prev filters and current array present but all search values are empty,
      then no need to make API call
    */
    if (!isEqual(currentConfig, this.prevConfig)) {
      if (isEmpty(this.prevConfig?.['filters'])) {
        // this check will be true for 2nd unchanged api call
        const allEmptyFilters = currentConfig.filters?.every(
          (field) => !field.search,
        ); // check if all filters passed are empty
        if (currentConfig.filters?.length && allEmptyFilters) {
          /* since there were no previous filters present(which mean no filters were applied/ has been reset already),
          and current filters are empty,
          api call made is duplicate call without change */
          this.tableLoader = false;
          return Promise.resolve(this.data);
        }
      }

      this.prevConfig = cloneDeep(currentConfig);
    }
    return super.getElements();
  }

  protected emitOnChanged(action: string): void {
    // https://github.com/akveo/ng2-smart-table/issues/131#issuecomment-349625947
    this.getElements()
      .then((elements) => {
        this.onChangedSource.next({
          action: action,
          elements: elements,
          paging: this.getPaging(),
          filter: this.getFilter(),
          sort: this.getSort(),
        });
      })
      .catch((error) => {
        this.tableLoader = false;
        this.errorHandler(error);
        this.onChangedSource.next({
          error: error,
          action: action,
          elements: [],
          paging: this.getPaging(),
          filter: this.getFilter(),
          sort: this.getSort(),
        });
      });
  }
}
